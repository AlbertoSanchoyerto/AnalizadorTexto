#makelist AnalizadorTexto
cmake_minimum_required(VERSION 3.16)

project(AnalizadorNLP VERSION 1.0 LANGUAGES CXX)

# Estándar de C++
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Carpetas
include_directories(include)

file(GLOB SOURCES
    src/*.cpp
)

add_executable(analizador
    ${SOURCES}
    main.cpp
)

# Opcional: warnings útiles
if (MSVC)
    target_compile_options(analizador PRIVATE /W4)
else()
    target_compile_options(analizador PRIVATE -Wall -Wextra -Wpedantic)
endif()
